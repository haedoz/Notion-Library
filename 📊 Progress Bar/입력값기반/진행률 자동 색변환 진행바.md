### 진행률에 따라 자동으로 색이 바뀌는 진행바
><img width="722" height="243" alt="스크린샷 2026-07-09 오후 3 29 34" src="https://github.com/user-attachments/assets/e094d073-8c5f-421a-bab1-0f5a5d69b6e8" />
```
let(
    fillIcon, "▓",
    emptyIcon, "░",
    current, toNumber(prop("읽고 있는 장수")),
    target, toNumber(prop("목표 장수")),
    if(
        empty(target) or target == 0,
        "목표량을 먼저 설정해주세요",
        let(
            pRaw, current / target,
            p, min(1, pRaw),
            filledCount, floor(p * 10),
            /* 🚦 진행률에 따라 색상을 자동으로 결정 */
            barColor, if(p < 0.3, "red", if(p < 0.5, "orange", "green")),
            bar,
            repeat(fillIcon, filledCount).style(barColor) +
            repeat(emptyIcon, 10 - filledCount).style("gray"),
            remain, max(0, target - current),
            bar + " " +
            format(round(pRaw * 100)) + "%" + " | " +
            format(remain) + "장 남음"
        )
    )
)

```
