## 📊 색상 변화 진행바

### 진행률에 따라 자동으로 색이 바뀌는 진행바
><img width="627" height="193" alt="스크린샷 2026-07-20 오후 12 51 14" src="https://github.com/user-attachments/assets/f2f4c358-3226-4a89-825d-f0aa68f10e00" />

```
let(
    /* 🎨 진행바 디자인 설정 */
    fillIcon, "▓",
    emptyIcon, "░",
    /* 📊 체크박스 개수 세기 */
    total, 7,
    current,
        toNumber(prop("기상")) +
        toNumber(prop("물 마시기")) +
        toNumber(prop("일정 체크")) +
        toNumber(prop("운동")) +
        toNumber(prop("가계부 작성")) +
        toNumber(prop("스킨케어")) +
        toNumber(prop("고양이 배변 치우기")),
    /* ⚙️ 진행바 계산 */
    p, current / total,
    filledCount, floor(p * 10),
    /* 🚦 진행률에 따라 색상을 자동으로 결정 */
    barColor, if(p < 0.3, "red", if(p < 0.7, "orange", "green")),
    bar,
    repeat(fillIcon, filledCount).style(barColor) +
    repeat(emptyIcon, 10 - filledCount).style("gray"),
    bar + " " + format(round(p * 100)) + "% (" + format(current) + "/" + format(total) + ")"
)
```
