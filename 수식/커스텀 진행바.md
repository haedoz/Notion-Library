### 01 미니멀 픽셀 네모 진행바
> <img width="719" height="241" alt="스크린샷 2026-07-08 오후 3 57 14" src="https://github.com/user-attachments/assets/cc50ab48-90ca-4554-9c1b-ec6f6e855c91" />

```
let(
    /* 🎨 1. 진행바 디자인 설정 (이 부분만 수정하세요!) */
    fillIcon, "■",   /* 채워진 모양 */
    emptyIcon, "□",  /* 비어있는 모양 */
    divider, " | ",  /* 텍스트 구분자 (예: |, ·) */

    /* 📊 2. 데이터 불러오기 */
    current, toNumber(prop("물 급수량")),
    target, toNumber(prop("목표 급수량")),

    /* ⚙️ 3. 진행바 계산 및 출력 (수정 불필요) */
    if(
        empty(target) or target == 0,
        "목표량을 먼저 설정해주세요",
        let(
            pRaw, current / target,
            p, min(1, pRaw),
            filledCount, floor(p * 10),

            bar,
			/* 🎨 색상 변경시 style("여기에 색상 영어로 작성해서 변경! ex) red")*/
            repeat(fillIcon, filledCount).style("blue") +
            repeat(emptyIcon, 10 - filledCount).style("gray"),

            remain, max(0, target - current),

            bar + " " +
            format(round(pRaw * 100)) + "%" + divider +
            format(remain) + "ml 남음"
        )
    )
)
```

### 응용편 : 진행률에 따라 자동으로 색이 바뀌는 진행바
><img width="717" height="238" alt="스크린샷 2026-07-08 오후 4 51 58" src="https://github.com/user-attachments/assets/ceffcd3a-b408-4040-a91e-50e3deac4db3" />

```
let(
    fillIcon, "▓",
    emptyIcon, "░",
    current, toNumber(prop("물 급수량")),
    target, toNumber(prop("목표 급수량")),
    if(
        empty(target) or target == 0,
        "목표량을 먼저 설정해주세요",
        let(
            pRaw, current / target,
            p, min(1, pRaw),
            filledCount, floor(p * 10),
            /* 🚦 진행률에 따라 색상을 자동으로 결정 */
            barColor, if(p < 0.3, "red", if(p < 0.7, "orange", "green")),
            bar,
            repeat(fillIcon, filledCount).style(barColor) +
            repeat(emptyIcon, 10 - filledCount).style("gray"),
            remain, max(0, target - current),
            bar + " " +
            format(round(pRaw * 100)) + "%" + " | " +
            format(remain) + "ml 남음"
        )
    )
)
```
